** Garbage Collection and IGC

We run on the IGC (incremental garbage collector) branch. Crank GC
thresholds sky-high during init, then reset. IGC-specific knobs
control step size and threshold separately.

#+begin_src emacs-lisp
  (setq gc-cons-threshold most-positive-fixnum
        gc-cons-percentage 0.6)

  (add-hook 'emacs-startup-hook
            (lambda ()
              (setq gc-cons-threshold (* 16 1024 1024)
                    gc-cons-percentage 0.1)))

  ;; IGC tuning (ignored on non-IGC builds)
  (setq igc-step-multiplier 2)
  (setq igc-cons-threshold (* 32 1024 1024))

  (add-hook 'focus-out-hook #'garbage-collect)
#+end_src

#+begin_src emacs-lisp
  (use-package ligature
  :ensure nil
  :config
  :config
  ;; Enable the "www" ligature in every possible major mode
  (ligature-set-ligatures 't '("www"))
  ;; Enable traditional ligature support in eww-mode, if the
  ;; `variable-pitch' face supports it
  (ligature-set-ligatures 'eww-mode '("ff" "fi" "ffi"))
  ;; Enable all Cascadia and Fira Code ligatures in programming modes
  (ligature-set-ligatures 'prog-mode
                        '(;; == === ==== => =| =>>=>=|=>==>> ==< =/=//=// =~
                          ;; =:= =!=
                          ("=" (rx (+ (or ">" "<" "|" "/" "~" ":" "!" "="))))
                          ;; ;; ;;;
                          (";" (rx (+ ";")))
                          ;; && &&&
                          ("&" (rx (+ "&")))
                          ;; !! !!! !. !: !!. != !== !~
                          ("!" (rx (+ (or "=" "!" "\." ":" "~"))))
                          ;; ?? ??? ?:  ?=  ?.
                          ("?" (rx (or ":" "=" "\." (+ "?"))))
                          ;; %% %%%
                          ("%" (rx (+ "%")))
                          ;; |> ||> |||> ||||> |] |} || ||| |-> ||-||
                          ;; |->>-||-<<-| |- |== ||=||
                          ;; |==>>==<<==<=>==//==/=!==:===>
                          ("|" (rx (+ (or ">" "<" "|" "/" ":" "!" "}" "\]"
                                          "-" "=" ))))
                          ;; \\ \\\ \/
                          ("\\" (rx (or "/" (+ "\\"))))
                          ;; ++ +++ ++++ +>
                          ("+" (rx (or ">" (+ "+"))))
                          ;; :: ::: :::: :> :< := :// ::=
                          (":" (rx (or ">" "<" "=" "//" ":=" (+ ":"))))
                          ;; // /// //// /\ /* /> /===:===!=//===>>==>==/
                          ("/" (rx (+ (or ">"  "<" "|" "/" "\\" "\*" ":" "!"
                                          "="))))
                          ;; .. ... .... .= .- .? ..= ..<
                          ("\." (rx (or "=" "-" "\?" "\.=" "\.<" (+ "\."))))
                          ;; -- --- ---- -~ -> ->> -| -|->-->>->--<<-|
                          ("-" (rx (+ (or ">" "<" "|" "~" "-"))))
                          ;; *> */ *)  ** *** ****
                          ("*" (rx (or ">" "/" ")" (+ "*"))))
                          ;; www wwww
                          ("w" (rx (+ "w")))
                          ;; <> <!-- <|> <: <~ <~> <~~ <+ <* <$ </  <+> <*>
                          ;; <$> </> <|  <||  <||| <|||| <- <-| <-<<-|-> <->>
                          ;; <<-> <= <=> <<==<<==>=|=>==/==//=!==:=>
                          ;; << <<< <<<<
                          ("<" (rx (+ (or "\+" "\*" "\$" "<" ">" ":" "~"  "!"
                                          "-"  "/" "|" "="))))
                          ;; >: >- >>- >--|-> >>-|-> >= >== >>== >=|=:=>>
                          ;; >> >>> >>>>
                          (">" (rx (+ (or ">" "<" "|" "/" ":" "=" "-"))))
                          ;; #: #= #! #( #? #[ #{ #_ #_( ## ### #####
                          ("#" (rx (or ":" "=" "!" "(" "\?" "\[" "{" "_(" "_"
                                       (+ "#"))))
                          ;; ~~ ~~~ ~=  ~-  ~@ ~> ~~>
                          ("~" (rx (or ">" "=" "-" "@" "~>" (+ "~"))))
                          ;; __ ___ ____ _|_ __|____|_
                          ("_" (rx (+ (or "_" "|"))))
                          ;; Fira code: 0xFF 0x12
                          ("0" (rx (and "x" (+ (in "A-F" "a-f" "0-9")))))
                          ;; Fira code:
                          "Fl"  "Tl"  "fi"  "fj"  "fl"  "ft"
                          ;; The few not covered by the regexps.
                          "{|"  "[|"  "]#"  "(*"  "}#"  "$>"  "^="))
  ;; Enables ligature checks globally in all buffers. You can also do it
  ;; per mode with `ligature-mode'.
  (global-ligature-mode t))

  (defun my-smarter-move-beginning-of-line (arg)
    "Move to first non-whitespace char, or column 0 if already there."
    (interactive "^p")
    (setq arg (or arg 1))
    (when (/= arg 1)
      (let ((line-move-visual nil))
        (forward-line (1- arg))))
    (let ((orig-point (point)))
      (back-to-indentation)
      (when (= orig-point (point))
        (move-beginning-of-line 1))))

  (defun my-duplicate-line ()
    "Duplicate the current line."
    (interactive)
    (let ((column (current-column)))
      (move-beginning-of-line 1)
      (kill-line)
      (yank)
      (open-line 1)
      (forward-line 1)
      (yank)
      (move-to-column column)))

  (defun my-move-line-up ()
    "Move current line up one line."
    (interactive)
    (transpose-lines 1)
    (forward-line -2))

  (defun my-move-line-down ()
    "Move current line down one line."
    (interactive)
    (forward-line 1)
    (transpose-lines 1)
    (forward-line -1))

** GPTel Tools
:PROPERTIES:
:header-args:emacs-lisp: :tangle init.el
:END:

Tools extend gptel so the LLM can take actions inside Emacs or read
external data during a conversation.  All tools here are registered
after gptel loads and are available to any backend that supports tool
use (Claude, GPT-4o, and capable local models via llama-cpp/Ollama).

*** Enabling tools

Tools are enabled globally with ~gptel-use-tools~.  To activate them
for a session, open the gptel transient menu with ~C-c g m~ and toggle
the tools you want under the "Tools" section.  You can also require
confirmation before any tool call fires from the same menu.

*** Available tools

| Tool                | Category   | What it does                                      |
|---------------------+------------+---------------------------------------------------|
| ~read_buffer~       | emacs      | Read the full contents of any live Emacs buffer   |
| ~read_file~         | filesystem | Read a file from disk by path                     |
| ~list_directory~    | filesystem | List files in a directory                         |
| ~list_project_files~| filesystem | List all files in the current project             |
| ~read_documentation~| emacs      | Look up docstrings for any Emacs symbol           |
| ~read_url~          | web        | Fetch and render a URL as plain text              |
| ~run_command~       | command    | Run a shell command (always asks for confirmation)|
| ~append_to_buffer~  | emacs      | Append text to a buffer, creating it if needed    |
| ~read_pdf~          | filesystem | Extract text from a PDF via pdf-tools             |

*** Typical workflows

- /Explain this buffer/: open a gptel chat, ask "explain the code in
  buffer foo.el" — Claude will call ~read_buffer~ automatically.

- /Project orientation/: ask "what files are in this project and what
  does the README say?" — Claude chains ~list_project_files~ then
  ~read_file~.

- /PDF Q&A/: with a paper on disk, ask "summarise
  ~/papers/foo.pdf~" — Claude calls ~read_pdf~ and summarises
  inline.

- /Documentation lookup/: ask "how does ~consult-ripgrep~ work?" —
  Claude calls ~read_documentation~ rather than relying on training
  data, giving you accurate docstrings.

- /Shell tasks/: ask "run the tests for this project" — Claude
  constructs the command and prompts you to confirm before running it.

*** Notes

- ~run_command~ has ~:confirm t~, meaning gptel will always show you
  the command and ask before executing it.  Do not remove this.
- ~read_pdf~ requires the ~epdfinfo~ server from pdf-tools to be
  running.  Opening any PDF in Emacs first is sufficient.
- ~list_project_files~ uses ~project.el~ and respects your
  ~.gitignore~ via ~project-vc-extra-root-markers~.
- Not all local models support tool use well.  If a model ignores
  tools, try a larger quant or switch to Claude/GPT-4o for
  tool-heavy sessions.

  (defun prot/keyboard-quit-dwim ()
    "Do-What-I-Mean behaviour for `keyboard-quit'."
    (interactive)
    (cond
     ((region-active-p)
      (deactivate-mark))
     ((derived-mode-p 'completion-list-mode)
      (delete-completion-window))
     ((> (minibuffer-depth) 0)
      (abort-recursive-edit))
     (t
      (keyboard-quit))))

  (defun my-project-find-file ()
    "Find file in current project, fallback to consult-find."
    (interactive)
    (if (and (fboundp 'project-current) (project-current))
        (call-interactively #'project-find-file)
      (if (fboundp 'consult-find)
          (consult-find default-directory)
        (call-interactively #'find-file))))

  (defun my-project-find-buffer ()
    "Project-aware buffer switch, fallback to consult-buffer."
    (interactive)
    (if (and (fboundp 'project-current) (project-current))
        (consult-project-buffer)
      (call-interactively #'consult-buffer)))
#+end_src

#+begin_src emacs-lisp
  (defun my-project-eat ()
  "Open eat in current project root with a project-named buffer."
  (interactive)
  (let* ((pr (project-current))
         (default-directory (if pr (project-root pr) default-directory))
         (name (if pr (project-name pr) "default"))
         (buf-name (format "*eat-%s*" name)))
    (if (get-buffer buf-name)
        (pop-to-buffer buf-name)
      (eat)
      (rename-buffer buf-name))))

  (defun my-project-compile ()
  "Compile in current project root with a project-named buffer."
  (interactive)
  (let* ((pr (project-current))
         (default-directory (if pr (project-root pr) default-directory))
         (name (if pr (project-name pr) "default"))
         (compilation-buffer-name-function
          (lambda (_mode) (format "*compilation-%s*" name))))
    (call-interactively #'compile)))

  (defun my-project-sql ()
    "Open SQL interactive in current project."
    (interactive)
    (let ((default-directory (or (and (project-current) (project-root (project-current)))
                                 default-directory)))
      (call-interactively #'sql-connect)))

  (define-key my-term-keymap (kbd "c") #'my-project-compile)
